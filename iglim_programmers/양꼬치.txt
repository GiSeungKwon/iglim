#include <iostream>

using namespace std;

int solution(int n, int k) {
    int answer = 0;
    
    cout << "양꼬치 " << n << "인분, 음료 " << k << "개" << endl;
    
    int service_n = n / 10;
    cout << "서비스 음료 " << service_n << "개" << endl;
    
    k = k - service_n;
    if (k < 0) {
        k = 0;
    }
    
    cout << "양꼬치 " << n << "인분, 음료 " << k << "개 계산 필요" << endl;
    answer = (n * 12000) + (k * 2000);
    
    return answer;
}
